Step 1: Create a Custom Annotation

First, create a single custom annotation, for example, 
@ValidPropertyValue, to be placed on the value field. 
This annotation will signal that a custom validation 
is required for the value based on the type.

import javax.validation.Constraint;
import javax.validation.Payload;
import java.lang.annotation.*;

@Documented
@Constraint(validatedBy = PropertyValueValidator.class)
@Target({ElementType.FIELD})
@Retention(RetentionPolicy.RUNTIME)
public @interface ValidPropertyValue {
    String message() default "Invalid value for the specified property type.";
    Class<?>[] groups() default {};
    Class<? extends Payload>[] payload() default {};
}

Step 2: Implement the Validator Logic

Now, implement the PropertyValueValidator class. This validator 
will check the type field of the Property object and apply the 
appropriate validation logic to the value string.

import javax.validation.ConstraintValidator;
import javax.validation.ConstraintValidatorContext;
import com.example.Property; // Adjust to your POJO's package
import java.math.BigDecimal;

public class PropertyValueValidator implements ConstraintValidator<ValidPropertyValue, String> {

    @Override
    public void initialize(ValidPropertyValue constraintAnnotation) {
        // Initialization if needed
    }

    @Override
    public boolean isValid(String value, ConstraintValidatorContext context) {
        // Retrieve the parent object (Property instance)
        Object parent = context.unwrap(Property.class);
        if (!(parent instanceof Property)) {
            return true; // No validation if the context is not a Property object
        }

        Property property = (Property) parent;
        String type = property.getType();

        // Custom validation based on the 'type'
        if ("Boolean".equalsIgnoreCase(type)) {
            // Check for valid Boolean values ("TRUE" or "FALSE")
            return "TRUE".equalsIgnoreCase(value) || "FALSE".equalsIgnoreCase(value);
        } else if ("Integer".equalsIgnoreCase(type)) {
            // Check if the value is a valid Integer
            try {
                Integer.parseInt(value);
                return true;
            } catch (NumberFormatException e) {
                return false;
            }
        } else if ("BigDecimal".equalsIgnoreCase(type)) {
            // Check if the value is a valid BigDecimal
            try {
                new BigDecimal(value);
                return true;
            } catch (NumberFormatException e) {
                return false;
            }
        }
        
        // If the type is not one of the special cases, the value is considered valid
        return true;
    }
}

Step 3: Add the Annotation to Your POJO

Update your Property POJO by adding the new @ValidPropertyValue 
annotation to the value field.

public class Property {
    private String key;
    private String type;

    @ValidPropertyValue
    private String value;

    public Property(String key, String type, String value) {
        this.key = key;
        this.type = type;
        this.value = value;
    }
    // Getters and setters
}

Step 4: Perform the Validation

Finally, perform the validation using a Validator instance 
before saving the object to the database. The validation logic 
remains the same as in the previous example, but now it will 
handle all the different data types.

import javax.validation.ConstraintViolation;
import javax.validation.Validation;
import javax.validation.Validator;
import javax.validation.ValidatorFactory;
import java.util.Set;

public class PropertyService {

    private final Validator validator;

    public PropertyService() {
        ValidatorFactory factory = Validation.buildDefaultValidatorFactory();
        this.validator = factory.getValidator();
    }

    public void saveProperty(Property property) {
        Set<ConstraintViolation<Property>> violations = validator.validate(property);

        if (!violations.isEmpty()) {
            // Handle validation errors
            for (ConstraintViolation<Property> violation : violations) {
                System.err.println("Validation Error: " + violation.getMessage() + " for property value '" + property.getValue() + "'");
            }
            throw new IllegalArgumentException("Property validation failed.");
        }

        // If validation passes, proceed with your Mybatis call to save the property
        System.out.println("Property is valid. Saving to database.");
        // mybatisMapper.insert(property);
    }

    public static void main(String[] args) {
        PropertyService service = new PropertyService();
        
        // Valid cases
        service.saveProperty(new Property("some_boolean", "Boolean", "TRUE"));
        service.saveProperty(new Property("some_integer", "Integer", "20"));
        service.saveProperty(new Property("some_decimal", "BigDecimal", "1.5"));
        
        System.out.println("\n--- Testing Invalid Cases ---");
        
        // Invalid cases
        try {
            service.saveProperty(new Property("invalid_boolean", "Boolean", "not_a_boolean"));
        } catch (IllegalArgumentException e) {
            System.err.println("Caught expected exception: " + e.getMessage());
        }
        
        try {
            service.saveProperty(new Property("invalid_integer", "Integer", "30.5"));
        } catch (IllegalArgumentException e) {
            System.err.println("Caught expected exception: " + e.getMessage());
        }
        
        try {
            service.saveProperty(new Property("invalid_decimal", "BigDecimal", "not_a_number"));
        } catch (IllegalArgumentException e) {
            System.err.println("Caught expected exception: " + e.getMessage());
        }
    }
}
