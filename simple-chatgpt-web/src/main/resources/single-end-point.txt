
@RestController
@RequestMapping("/mybatis/data")
public class GenericDataController {

    private static final Logger logger = LogManager.getLogger(GenericDataController.class);

    private final UserService userService;
    private final ItemService itemService;
    private final ProductService productService;
    // Add more services here...

    @Autowired
    public GenericDataController(UserService userService,
                                 ItemService itemService,
                                 ProductService productService) {
        this.userService = userService;
        this.itemService = itemService;
        this.productService = productService;
    }

    //------------------------------
    // Add
    @PostMapping("/add")
    public ResponseEntity<Response<?>> add(@RequestParam String type, @RequestBody Map<String, Object> payload) {
        logger.debug("Add request type: {}, payload: {}", type, payload);
        Object result;

        switch (type.toLowerCase()) {
            case "user":
                User user = mapTo(payload, User.class);
                result = userService.save(user);
                break;
            case "item":
                Item item = mapTo(payload, Item.class);
                result = itemService.save(item);
                break;
            case "product":
                Product product = mapTo(payload, Product.class);
                result = productService.save(product);
                break;
            default:
                return ResponseEntity.badRequest().body(Response.error("Unknown type", null, 400));
        }

        return ResponseEntity.ok(Response.success(type + " saved successfully", result, 200));
    }

    //------------------------------
    // Update
    @PutMapping("/{id}")
    public ResponseEntity<Response<?>> update(@RequestParam String type, @PathVariable Long id, @RequestBody Map<String, Object> payload) {
        logger.debug("Update request type: {}, id: {}, payload: {}", type, id, payload);
        Object result;

        switch (type.toLowerCase()) {
            case "user":
                User user = mapTo(payload, User.class);
                user.setId(id);
                result = userService.save(user);
                break;
            case "item":
                Item item = mapTo(payload, Item.class);
                item.setId(id);
                result = itemService.save(item);
                break;
            case "product":
                Product product = mapTo(payload, Product.class);
                product.setId(id);
                result = productService.save(product);
                break;
            default:
                return ResponseEntity.badRequest().body(Response.error("Unknown type", null, 400));
        }

        return ResponseEntity.ok(Response.success(type + " updated successfully", result, 200));
    }

    //------------------------------
    // Get by ID
    @GetMapping("/{id}")
    public ResponseEntity<Response<?>> get(@RequestParam String type, @PathVariable Long id) {
        Object result;

        switch (type.toLowerCase()) {
            case "user":
                result = userService.findById(id);
                break;
            case "item":
                result = itemService.findById(id);
                break;
            case "product":
                result = productService.findById(id);
                break;
            default:
                return ResponseEntity.badRequest().body(Response.error("Unknown type", null, 400));
        }

        if (result == null) {
            return ResponseEntity.status(404).body(Response.error(type + " not found", null, 404));
        }

        return ResponseEntity.ok(Response.success(type + " retrieved successfully", result, 200));
    }

    //------------------------------
    // Get all (optional paging)
    @GetMapping("/all")
    public ResponseEntity<Response<?>> getAll(@RequestParam String type) {
        Object result;

        switch (type.toLowerCase()) {
            case "user":
                result = userService.findAll();
                break;
            case "item":
                result = itemService.findAll();
                break;
            case "product":
                result = productService.findAll();
                break;
            default:
                return ResponseEntity.badRequest().body(Response.error("Unknown type", null, 400));
        }

        return ResponseEntity.ok(Response.success(type + " list retrieved", result, 200));
    }

    //------------------------------
    // Utility to convert Map to POJO (you can use Jackson ObjectMapper)
    private <T> T mapTo(Map<String, Object> map, Class<T> clazz) {
        return new com.fasterxml.jackson.databind.ObjectMapper().convertValue(map, clazz);
    }
}

✅ Features

Single endpoint for multiple types using a type query parameter.

Returns typed JSON wrapped in Response<T>.

Handles add, update, get by ID, get all.

Uses Map<String, Object> in @RequestBody and converts dynamically to the proper POJO.

Easy to extend for new types: just add the service in the constructor and add a case in switch.

const API_DATA = '/mybatis/data';
const TYPE_CONFIG = 'config';

async function configExamples() {
    console.log('--- CONFIG ---');

    // Add
    let res = await fetch(`${API_DATA}/add?type=${TYPE_CONFIG}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },
        body: JSON.stringify({ name: 'ConfigA', value: 'ValueA' })
    });
    console.log('Config add:', await res.json());

    // Update by ID
    const configId = 1;
    res = await fetch(`${API_DATA}/${configId}?type=${TYPE_CONFIG}`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },
        body: JSON.stringify({ name: 'ConfigA-Updated', value: 'ValueA2' })
    });
    console.log('Config update by ID:', await res.json());

    // Update Map
    res = await fetch(`${API_DATA}/update?type=${TYPE_CONFIG}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },
        body: JSON.stringify({ key: 'ConfigA', value: 'ValueMap' })
    });
    console.log('Config update map:', await res.json());

    // All
    res = await fetch(`${API_DATA}/all?type=${TYPE_CONFIG}`, { headers: { 'Accept': 'application/json' } });
    console.log('Config all:', await res.json());

    // Paged
    res = await fetch(`${API_DATA}/paged?type=${TYPE_CONFIG}&page=1&size=5`, { headers: { 'Accept': 'application/json' } });
    console.log('Config paged:', await res.json());
}
