
import org.apache.logging.log4j.LogManager;
import org.apache.logging.log4j.Logger;

import java.util.*;
import java.util.concurrent.*;

public class MultiAgentTaskSystem {

    private static final Logger logger = LogManager.getLogger(MultiAgentTaskSystem.class);

    public static void main(String[] args) {
        logger.debug("main called");

        TaskQueue taskQueue = new TaskQueue();
        AgentRegistry agentRegistry = new AgentRegistry();

        // create tasks
        List<Task> tasks = Arrays.asList(
            new Task("T1", "Process data segment A"),
            new Task("T2", "Analyze image set B"),
            new Task("T3", "Generate summary report C")
        );
        logger.debug("main tasks={}", tasks);

        // register agents
        Agent agent1 = new Agent("Agent-Alpha");
        Agent agent2 = new Agent("Agent-Beta");
        agentRegistry.register(agent1);
        agentRegistry.register(agent2);

        logger.debug("main agentRegistry={}", agentRegistry);

        // enqueue tasks
        taskQueue.enqueue(tasks);
        logger.debug("main taskQueue={}", taskQueue);

        // run system
        MultiAgentController controller = new MultiAgentController(agentRegistry, taskQueue);
        controller.executeAll();
    }
}

class Task {
    private final String id;
    private final String description;

    public Task(String id, String description) {
        this.id = id;
        this.description = description;
    }

    public String getId() { return id; }
    public String getDescription() { return description; }

    @Override
    public String toString() {
        return "Task{" + "id='" + id + '\'' + ", description='" + description + '\'' + '}';
    }
}

class Agent {
    private static final Logger logger = LogManager.getLogger(Agent.class);

    private final String name;

    public Agent(String name) {
        this.name = name;
        logger.debug("Agent constructor name={}", name);
    }

    public String getName() { return name; }

    public void perform(Task task) {
        logger.debug("perform task={} by agent={}", task, name);
        // simulate work
        try {
            Thread.sleep(500);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            logger.error("Agent {} was interrupted while performing {}", name, task, e);
        }
        logger.debug("Agent {} completed task={}", name, task);
    }

    @Override
    public String toString() {
        return "Agent{" + "name='" + name + '\'' + '}';
    }
}

class AgentRegistry {
    private static final Logger logger = LogManager.getLogger(AgentRegistry.class);

    private final List<Agent> agents = new ArrayList<>();

    public void register(Agent agent) {
        logger.debug("register agent={}", agent);
        agents.add(agent);
    }

    public List<Agent> getAgents() {
        logger.debug("getAgents called");
        return Collections.unmodifiableList(agents);
    }

    @Override
    public String toString() {
        return "AgentRegistry{" + "agents=" + agents + '}';
    }
}

class TaskQueue {
    private static final Logger logger = LogManager.getLogger(TaskQueue.class);

    private final Queue<Task> queue = new ConcurrentLinkedQueue<>();

    public void enqueue(List<Task> tasks) {
        logger.debug("enqueue tasks={}", tasks);
        queue.addAll(tasks);
    }

    public Task dequeue() {
        logger.debug("dequeue called");
        return queue.poll();
    }

    @Override
    public String toString() {
        return "TaskQueue{" + "remaining=" + queue.size() + '}';
    }
}

class MultiAgentController {
    private static final Logger logger = LogManager.getLogger(MultiAgentController.class);

    private final AgentRegistry agentRegistry;
    private final TaskQueue taskQueue;
    private final ExecutorService executor;

    public MultiAgentController(AgentRegistry agentRegistry, TaskQueue taskQueue) {
        logger.debug("MultiAgentController constructor agentRegistry={}", agentRegistry);
        logger.debug("MultiAgentController constructor taskQueue={}", taskQueue);

        this.agentRegistry = agentRegistry;
        this.taskQueue = taskQueue;
        this.executor = Executors.newFixedThreadPool(agentRegistry.getAgents().size());
    }

    public void executeAll() {
        logger.debug("executeAll called");

        for (Agent agent : agentRegistry.getAgents()) {
            executor.submit(() -> {
                Task task;
                while ((task = taskQueue.dequeue()) != null) {
                    logger.debug("Agent {} picked up task={}", agent.getName(), task);
                    agent.perform(task);
                }
                logger.debug("Agent {} found no more tasks", agent.getName());
            });
        }

        executor.shutdown();
        try {
            executor.awaitTermination(10, TimeUnit.MINUTES);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            logger.error("executeAll interrupted", e);
        }
        logger.debug("executeAll completed");
    }
}

What it does

You define a team (“crew”) of agents, each with a role, goal, and back-story (so the agent “persona” is specified). 
Medium
+1

For example:

A Planner agent whose job is to research and create an outline for a blog article on a given topic. 
Medium

A Writer agent whose job is to take that outline and produce a draft article (in markdown) using SEO keywords etc. 
Medium

An Editor agent whose job is to review the article, align it with style / brand, fix grammar, ensure balance. 
Medium

You create Task objects: each task has a description (what to do), an expected output, and is assigned to a specific agent. 
Medium

Then you orchestrate the tasks in sequence: the planner runs first, producing an outline; then the writer uses that outline; then the editor finalizes. So you have a multi-step workflow.

The example emphasises core design principles:

Role-playing: giving agents clear identity and purpose.

Granularity / Focus: breaking a large job (write article) into clear sub-tasks.

Cooperation / Sequence: agents passing results to one another.

Tools, Memory, Guardrails: although in this example tools may be simpler, the course emphasises that real agent systems integrate memory (short/long/shared), external tools (search, file-read, etc), and guardrails (to avoid infinite loops, hallucinations). 
Medium
+1

It shows how orchestrating multiple specialised agents outperforms simply asking one monolithic agent (LLM) to do everything.

import org.apache.logging.log4j.LogManager;
import org.apache.logging.log4j.Logger;

import java.util.*;

public class AgentCrewExample {

    private static final Logger logger = LogManager.getLogger(AgentCrewExample.class);

    public static void main(String[] args) {
        logger.debug("main called");

        // set topic (input to the crew)
        String topic = "AI multi-agent systems";

        logger.debug("main topic={}", topic);

        // Instantiate agents
        Agent planner = new Agent("Content Planner", 
                                  "Plan engaging and factually accurate content on " + topic,
                                  "You're working on planning a blog article about the topic: " + topic + ".");
        logger.debug("main planner={}", planner);

        Agent writer = new Agent("Content Writer", 
                                 "Write insightful and factually accurate opinion piece about the topic: " + topic,
                                 "You're working on writing a new opinion piece about the topic: " + topic + ".");
        logger.debug("main writer={}", writer);

        Agent editor = new Agent("Editor", 
                                 "Edit a given blog post to align with the writing style of the organization.",
                                 "You are an editor who receives a blog post from the Content Writer.");
        logger.debug("main editor={}", editor);

        // Create tasks
        Task planTask = new Task(planner,
                                 "1. Prioritize latest trends, key players and noteworthy news on " + topic + ".\n"
                                 + "2. Identify the target audience, considering their interests and pain-points.\n"
                                 + "3. Develop a detailed content outline including an introduction, key points, and a call to action.\n"
                                 + "4. Include SEO keywords and relevant data or sources.",
                                 "A comprehensive content plan document with an outline, audience analysis, SEO keywords, and resources.");
        logger.debug("main planTask={}", planTask);

        Task writeTask = new Task(writer,
                                  "1. Use the content plan to craft a compelling blog post on " + topic + ".\n"
                                  + "2. Incorporate SEO keywords naturally.\n"
                                  + "3. Sections/Subtitles properly named in an engaging manner.\n"
                                  + "4. Ensure the post is structured with engaging introduction, insightful body, summarizing conclusion.\n"
                                  + "5. Proof-read for grammar and alignment with the brand's voice.",
                                  "A well-written blog post in markdown format, ready for publication, each section with 2-3 paragraphs.");
        logger.debug("main writeTask={}", writeTask);

        Task editTask = new Task(editor,
                                 "Proof-read the given blog post for grammatical errors and alignment with the brand's voice.",
                                 "A well-written blog post in markdown format, ready for publication, each section with 2-3 paragraphs.");
        logger.debug("main editTask={}", editTask);

        // Create workflow (crew)
        List<Task> workflow = Arrays.asList(planTask, writeTask, editTask);
        logger.debug("main workflow={}", workflow);

        // Execute workflow
        CrewController controller = new CrewController(workflow);
        controller.execute(topic);

        logger.debug("main completed");
    }
}

class Agent {
    private static final Logger logger = LogManager.getLogger(Agent.class);

    private final String role;
    private final String goal;
    private final String backstory;

    public Agent(String role, String goal, String backstory) {
        logger.debug("Agent constructor role={}, goal={}, backstory={}", role, goal, backstory);
        this.role = role;
        this.goal = goal;
        this.backstory = backstory;
    }

    public String getRole() {
        logger.debug("getRole called");
        return role;
    }

    public String getGoal() {
        logger.debug("getGoal called");
        return goal;
    }

    public String getBackstory() {
        logger.debug("getBackstory called");
        return backstory;
    }

    public String perform(Task task, String input) {
        logger.debug("perform called agent={}, task={}, input={}", this, task, input);
        // In real system: call LLM / tool to produce output. Here we simulate.
        String result = String.format("[Agent %s generated output for task \"%s\" with input \"%s\"]", role, task.getDescription(), input);
        logger.debug("perform result={}", result);
        return result;
    }

    @Override
    public String toString() {
        return "Agent{" + "role='" + role + '\'' + '}';
    }
}

class Task {
    private static final Logger logger = LogManager.getLogger(Task.class);

    private final Agent agent;
    private final String description;
    private final String expectedOutput;

    public Task(Agent agent, String description, String expectedOutput) {
        logger.debug("Task constructor agent={}, description={}, expectedOutput={}", agent, description, expectedOutput);
        this.agent = agent;
        this.description = description;
        this.expectedOutput = expectedOutput;
    }

    public Agent getAgent() {
        logger.debug("getAgent called");
        return agent;
    }

    public String getDescription() {
        logger.debug("getDescription called");
        return description;
    }

    public String getExpectedOutput() {
        logger.debug("getExpectedOutput called");
        return expectedOutput;
    }

    @Override
    public String toString() {
        return "Task{" + "agent=" + agent + ", description='" + description + '\'' + '}';
    }
}

class CrewController {
    private static final Logger logger = LogManager.getLogger(CrewController.class);

    private final List<Task> tasks;

    public CrewController(List<Task> tasks) {
        logger.debug("CrewController constructor tasks={}", tasks);
        this.tasks = tasks;
    }

    public void execute(String initialInput) {
        logger.debug("execute called initialInput={}", initialInput);
        String currentInput = initialInput;
        for (Task task : tasks) {
            Agent agent = task.getAgent();
            logger.debug("execute: starting task={} by agent={}", task, agent);
            String output = agent.perform(task, currentInput);
            logger.debug("execute: output from agent={} output={}", agent, output);
            currentInput = output;
        }
        logger.debug("execute completed");
    }
}
