We’ll give the model an unstructured text — e.g. a product review —
and ask it to return structured JSON data with fields like {"sentiment": "...", "product": "...", "summary": "..."}

import com.openai.client.OpenAI;
import com.openai.models.ChatCompletion;
import com.openai.models.ChatCompletionChoice;
import com.openai.models.ChatCompletionMessage;
import org.apache.logging.log4j.LogManager;
import org.apache.logging.log4j.Logger;

import java.util.Arrays;

public class ReviewExtractorService {

    private static final Logger logger = LogManager.getLogger(ReviewExtractorService.class);

    public static void main(String[] args) {
        ReviewExtractorService service = new ReviewExtractorService();
        service.extractReviewData();
    }

    private void extractReviewData() {
        logger.debug("extractReviewData called");

        // Initialize OpenAI client
        OpenAI client = new OpenAI.Builder()
                .apiKey(System.getenv("OPENAI_API_KEY"))
                .build();
        logger.debug("extractReviewData client initialized");

        // Example input text (like from user)
        String reviewText = "I bought the iPhone 15 last week and I absolutely love it! " +
                "The camera is stunning but battery life could be better.";
        logger.debug("extractReviewData reviewText={}", reviewText);

        // Create messages for ChatGPT
        ChatCompletion completion = client.chatCompletions().create(c -> c
                .model("gpt-4-turbo")
                .messages(Arrays.asList(
                        ChatCompletionMessage.builder()
                                .role("system")
                                .content("You are an AI that extracts structured data in JSON format from text reviews.")
                                .build(),
                        ChatCompletionMessage.builder()
                                .role("user")
                                .content(
                                        "Extract the following information as JSON: " +
                                                "product name, sentiment (positive/negative/neutral), and one-sentence summary.\n\n" +
                                                "Text: " + reviewText
                                )
                                .build()
                ))
        );
        logger.debug("extractReviewData completion request created");

        // Parse and log output
        for (ChatCompletionChoice choice : completion.choices()) {
            String result = choice.message().content();
            logger.debug("extractReviewData result={}", result);
            System.out.println(result);
        }
    }
}

What You Learn Here (same as in the course)

System + User roles: used to control tone and behavior.

Structured output prompts: asking explicitly for JSON makes it easier for your app to parse.

Logging for every param and step (as per your rule set).

import com.openai.client.OpenAI;

import com.openai.models.ChatCompletionChoice;

    public void summarizeText(String text) {
        logger.debug("summarizeText called");
        logger.debug("summarizeText text={}", text);

        OpenAI client = new OpenAI.Builder()
                .apiKey(System.getenv("OPENAI_API_KEY"))
                .build();
        logger.debug("summarizeText client initialized");

        ChatCompletion completion = client.chatCompletions().create(c -> c
                .model("gpt-4-turbo")
                .messages(Arrays.asList(
                        ChatCompletionMessage.builder()
                                .role("system")

                                .build(),
                        ChatCompletionMessage.builder()
                                .role("user")
                                .content("Summarize this in one sentence: " + text)
                                .build()

            logger.debug("summarizeText result={}", choice.message().content());

Few-Shot Example (Teach by Example)

(Technique: Few-shot prompting)

This is a core concept in the course: showing the model examples to guide it

import com.openai.client.OpenAI;
import com.openai.models.ChatCompletion;
import com.openai.models.ChatCompletionChoice;

import java.util.Arrays;

    private static final Logger logger = LogManager.getLogger(FewShotExampleService.class);

    public void inferSentiment(String text) {
        logger.debug("inferSentiment called");
        logger.debug("inferSentiment text={}", text);

                                .role("user")
                                .content("Text: \"I love the new design!\" → Sentiment: Positive")
                                .build(),
                        ChatCompletionMessage.builder()

Step-by-Step Reasoning Example

(Technique: Chain-of-thought prompting)

import com.openai.client.OpenAI;
import com.openai.models.ChatCompletion;
import com.openai.models.ChatCompletionChoice;
import com.openai.models.ChatCompletionMessage;
import org.apache.logging.log4j.LogManager;
import org.apache.logging.log4j.Logger;

import java.util.Arrays;

public class ReasoningService {

    private static final Logger logger = LogManager.getLogger(ReasoningService.class);

    public void solveLogicProblem() {
        logger.debug("solveLogicProblem called");

        OpenAI client = new OpenAI.Builder()
                .apiKey(System.getenv("OPENAI_API_KEY"))
                .build();
        logger.debug("solveLogicProblem client initialized");

        String problem = "If John is older than Mary and Mary is older than Sam, who is the oldest?";
        logger.debug("solveLogicProblem problem={}", problem);

        ChatCompletion completion = client.chatCompletions().create(c -> c
                .model("gpt-4-turbo")
                .messages(Arrays.asList(
                        ChatCompletionMessage.builder()
                                .role("system")
                                .content("You are a reasoning assistant. Think step by step before answering.")
                                .build(),
                        ChatCompletionMessage.builder()
                                .role("user")
                                .content(problem + " Let's reason this out step by step.")
                                .build()
                ))
        );
        logger.debug("solveLogicProblem completion request created");

        for (ChatCompletionChoice choice : completion.choices()) {
            logger.debug("solveLogicProblem result={}", choice.message().content());
            System.out.println(choice.message().content());
        }
    }

    public static void main(String[] args) {
        new ReasoningService().solveLogicProblem();
    }
}
