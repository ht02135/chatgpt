    @Bean
    public SecurityFilterChain securityFilterChain(HttpSecurity http) throws Exception {
        logger.debug("securityFilterChain called");

        http
            .csrf(csrf -> csrf.disable())
            .sessionManagement(session -> session.sessionCreationPolicy(SessionCreationPolicy.STATELESS))
            .authorizeHttpRequests(auth -> auth
                .requestMatchers("/api/admin/**").hasRole("ADMIN_ROLE_GROUP")
                .requestMatchers("/api/user/**").hasRole("USER_ROLE_GROUP")
                .anyRequest().authenticated()
            )
            .addFilterBefore(jwtAuthenticationFilter(), UsernamePasswordAuthenticationFilter.class);

        return http.build();
    }

    @Bean
    public JwtAuthenticationFilter jwtAuthenticationFilter() {
        logger.debug("jwtAuthenticationFilter called");
        return new JwtAuthenticationFilter();
    }
}

@Component
public class JwtAuthenticationFilter extends OncePerRequestFilter {

    private static final Logger logger = LogManager.getLogger(JwtAuthenticationFilter.class);
    private final JwtService jwtService;
    private final UserDetailsService userDetailsService;

    public JwtAuthenticationFilter(JwtService jwtService, UserDetailsService userDetailsService) {
        logger.debug("JwtAuthenticationFilter constructor called");
        this.jwtService = jwtService;
        this.userDetailsService = userDetailsService;
    }

    @Override
    protected void doFilterInternal(HttpServletRequest request,
                                    HttpServletResponse response,
                                    FilterChain filterChain)
                                    throws ServletException, IOException {
        logger.debug("doFilterInternal called");
        logger.debug("request.getRequestURI={}", request.getRequestURI());

        String authHeader = request.getHeader("Authorization");
        logger.debug("authHeader={}", authHeader);

        if (authHeader == null || !authHeader.startsWith("Bearer ")) {
            logger.debug("No JWT token found in header");
            filterChain.doFilter(request, response);
            return;
        }

        String jwt = authHeader.substring(7);
        logger.debug("jwt={}", jwt);

        String username = jwtService.extractUsername(jwt);
        logger.debug("username extracted from jwt={}", username);

        if (username != null && SecurityContextHolder.getContext().getAuthentication() == null) {
            UserDetails userDetails = userDetailsService.loadUserByUsername(username);
            logger.debug("userDetails loaded={}", userDetails);

            if (jwtService.isTokenValid(jwt, userDetails)) {
                UsernamePasswordAuthenticationToken authToken = new UsernamePasswordAuthenticationToken(
                        userDetails,
                        null,
                        userDetails.getAuthorities()
                );
                logger.debug("authToken created={}", authToken);
                SecurityContextHolder.getContext().setAuthentication(authToken);
            }
        }
        filterChain.doFilter(request, response);
    }
}

how will you handle assigning rolegroup to page??

You want:

Each page (or API endpoint) to be associated with one or more RoleGroups.

Access to be granted if user’s RoleGroups intersect with the page’s RoleGroups.

Extensible to add new RoleGroups (e.g., MANAGER_ROLE_GROUP) later.

Table	Columns	Description
role_group	id, name	e.g. ADMIN_ROLE_GROUP, USER_ROLE_GROUP
role	id, name, role_group_id	e.g. ADMIN_ROLE, belongs to ADMIN_ROLE_GROUP
page_access	id, url_pattern, method, role_group_id	e.g. /api/admin/**, GET, ADMIN_ROLE_GROUP

During startup, or every few minutes (cache refresh), load all PageAccess mappings into memory.

When an HTTP request comes in:

Your JwtAuthenticationFilter authenticates the user and sets Authentication with user’s RoleGroups.

A custom AuthorizationFilter checks if the request URI matches any configured page pattern.

Then it verifies if the user’s RoleGroups intersect with allowed RoleGroups for that page.

@Service
public class PageAccessService {
    private static final Logger logger = LogManager.getLogger(PageAccessService.class);

    private final PageAccessRepository pageAccessRepository;

    public PageAccessService(PageAccessRepository pageAccessRepository) {
        logger.debug("PageAccessService constructor called");
        this.pageAccessRepository = pageAccessRepository;
    }

    public List<PageAccess> getAllPageAccess() {
        logger.debug("getAllPageAccess called");
        List<PageAccess> result = pageAccessRepository.findAll();
        logger.debug("getAllPageAccess result={}", result);
        return result;
    }

    public Set<RoleGroup> getAllowedRoleGroups(String path, String method) {
        logger.debug("getAllowedRoleGroups path={}", path);
        logger.debug("getAllowedRoleGroups method={}", method);

        List<PageAccess> all = pageAccessRepository.findAll();
        for (PageAccess pa : all) {
            if (path.matches(pa.getUrlPattern()) && pa.getMethod().equalsIgnoreCase(method)) {
                logger.debug("Matched PageAccess={}", pa);
                return Set.of(pa.getRoleGroup());
            }
        }
        logger.debug("No match found for path={}", path);
        return Set.of();
    }
}

@Component
public class RoleGroupAuthorizationFilter extends OncePerRequestFilter {

    private static final Logger logger = LogManager.getLogger(RoleGroupAuthorizationFilter.class);
    private final PageAccessService pageAccessService;

    public RoleGroupAuthorizationFilter(PageAccessService pageAccessService) {
        logger.debug("RoleGroupAuthorizationFilter constructor called");
        this.pageAccessService = pageAccessService;
    }

    @Override
    protected void doFilterInternal(HttpServletRequest request,
                                    HttpServletResponse response,
                                    FilterChain filterChain)
                                    throws ServletException, IOException {
        logger.debug("doFilterInternal called");
        logger.debug("requestURI={}", request.getRequestURI());
        logger.debug("method={}", request.getMethod());

        Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
        if (authentication == null) {
            logger.debug("No authentication found");
            response.sendError(HttpServletResponse.SC_UNAUTHORIZED, "Unauthenticated");
            return;
        }

        Set<String> userRoleGroups = authentication.getAuthorities().stream()
                .map(GrantedAuthority::getAuthority)
                .collect(Collectors.toSet());
        logger.debug("userRoleGroups={}", userRoleGroups);

        Set<RoleGroup> allowedGroups = pageAccessService.getAllowedRoleGroups(
                request.getRequestURI(), request.getMethod());
        logger.debug("allowedGroups={}", allowedGroups);

        boolean allowed = allowedGroups.stream()
                .anyMatch(rg -> userRoleGroups.contains(rg.getName()));
        logger.debug("allowed={}", allowed);

        if (!allowed) {
            response.sendError(HttpServletResponse.SC_FORBIDDEN, "Access Denied");
            return;
        }

        filterChain.doFilter(request, response);
    }
}

@Configuration
@EnableWebSecurity
public class SecurityConfig {
    private static final Logger logger = LogManager.getLogger(SecurityConfig.class);

    private final JwtAuthenticationFilter jwtAuthenticationFilter;
    private final RoleGroupAuthorizationFilter roleGroupAuthorizationFilter;

    public SecurityConfig(JwtAuthenticationFilter jwtAuthenticationFilter,
                          RoleGroupAuthorizationFilter roleGroupAuthorizationFilter) {
        logger.debug("SecurityConfig constructor called");
        this.jwtAuthenticationFilter = jwtAuthenticationFilter;
        this.roleGroupAuthorizationFilter = roleGroupAuthorizationFilter;
    }

    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        logger.debug("filterChain called");

        http
            .csrf(csrf -> csrf.disable())
            .sessionManagement(session -> session.sessionCreationPolicy(SessionCreationPolicy.STATELESS))
            .authorizeHttpRequests(auth -> auth.anyRequest().permitAll()) // authorization handled by custom filter
            .addFilterBefore(jwtAuthenticationFilter, UsernamePasswordAuthenticationFilter.class)
            .addFilterAfter(roleGroupAuthorizationFilter, JwtAuthenticationFilter.class);

        return http.build();
    }
}
