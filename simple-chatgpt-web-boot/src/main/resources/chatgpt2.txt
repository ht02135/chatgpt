

import com.crewai.Agent;
import com.crewai.Task;
import com.crewai.Crew;
import com.crewai.ProcessType;

public class CustomerSupportCrew {

    private static final org.apache.logging.log4j.Logger logger =
        org.apache.logging.log4j.LogManager.getLogger(CustomerSupportCrew.class);

    public static void main(String[] args) {
        logger.debug("main called");

        // Create agents
        logger.debug("creating agent supportAgent");
        Agent supportAgent = Agent.builder()
            .name("support_agent")
            .role("Senior Customer Support Agent – handles inbound customer inquiries, resolves tickets")
            .goal("Resolve customer support inquiries with appropriate responses and escalate when needed")
            .tools( /* e.g., ticketingSystemTool, knowledgeBaseSearchTool */ )
            .build();

        logger.debug("supportAgent = {}", supportAgent);

        logger.debug("creating agent qaAgent");
        Agent qaAgent = Agent.builder()
            .name("support_quality_assurance_agent")
            .role("QA Support Agent – reviews resolved tickets for quality, adherence to SLA, correctness")
            .goal("Review completed tickets, provide feedback, flag issues for retraining or escalation")
            .tools( /* e.g., ticketAuditTool, feedbackLoggingTool */ )
            .build();

        logger.debug("qaAgent = {}", qaAgent);

        // Create tasks
        logger.debug("creating task inquiryResolution");
        Task inquiryResolution = Task.builder()
            .name("inquiry_resolution")
            .description("Agent handles customer inquiry end-to-end: read ticket, respond, close or escalate")
            .expectedOutput("Ticket resolved or escalated with resolution summary")
            .agent(supportAgent)
            // optionally task-level tools
            .tools(/* e.g., knowledgeBaseSearchTool */)
            .build();

        logger.debug("inquiryResolution = {}", inquiryResolution);

        logger.debug("creating task qualityAssuranceReview");
        Task qaReview = Task.builder()
            .name("quality_assurance_review")
            .description("QA Agent reviews resolved tickets for quality, logs feedback, and optionally triggers retraining")
            .expectedOutput("Feedback record for ticket, quality score, and retraining flag if needed")
            .agent(qaAgent)
            // optionally task-level tools
            .tools(/* e.g., feedbackLoggingTool */)
            .build();

        logger.debug("qualityAssuranceReview = {}", qaReview);

        // Create a crew to orchestrate
        logger.debug("creating crew supportCrew");
        Crew supportCrew = Crew.builder()
            .name("customer_support_crew")
            .agents(supportAgent, qaAgent)
            .tasks(inquiryResolution, qaReview)
            .process(ProcessType.SEQUENTIAL)  // first supportAgent executes, then QA
            .verbose(true)
            .build();

        logger.debug("supportCrew = {}", supportCrew);

        // Execute the workflow
        supportCrew.run();
    }
}
