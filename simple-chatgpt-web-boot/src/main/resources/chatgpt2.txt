
| Old XML Feature                       | New Java Config Equivalent                                      |
| ------------------------------------- | --------------------------------------------------------------- |
| `<job>` with steps & decision         | `@Configuration` class defining a `Job` bean with `StepBuilder` |
| `<CronTriggerBean>` + `<JobDetail>`   | `QuartzSchedulerConfig` with `JobDetail` and `Trigger`          |
| `<tasklet>`, `<chunk>`                | Modern `Tasklet` or `ChunkOrientedTasklet` beans                |
| `<property>`/`<ref>` wiring           | Spring Boot / `@Bean` dependency injection                      |
| MyBatis readers/writers               | Still usable — just injected via beans                          |
| XML imports (ftp.xml, fileNaming.xml) | Centralized config classes (`@Configuration`)                   |
| `PropertyPlaceholderConfigurer`       | Use `application.yml` / Spring `@Value` injection               |

| Old (XML)                  | New (Spring Batch 5.2.3)                                           |
| -------------------------- | ------------------------------------------------------------------ |
| `<job>` / `<step>` XML     | Java DSL via `JobBuilder`, `StepBuilder`                           |
| `CronTriggerBean` (Quartz) | `@Scheduled(cron = "...")`                                         |
| `jobDetail` beans          | Replaced with `JobLauncher` call                                   |
| XML listeners              | Implement `JobExecutionListener` / `StepExecutionListener` classes |
| `scope="step"` beans       | Annotate with `@StepScope` in Java config                          |

@Configuration
@EnableScheduling
@RequiredArgsConstructor
public class ChatGptExtractJobConfig {

    private static final Logger logger = LogManager.getLogger(ChatGptExtractJobConfig.class);

    private final JobRepository jobRepository;
    private final PlatformTransactionManager transactionManager;
    private final ChatGptExtractTasklet extractTasklet;
    private final EncryptFileTasklet encryptFileTasklet;
    private final FileTransferTasklet fileTransferTasklet;
    private final JobExecutionDecider decider;

    /*
      hung: ChatGptExtract modern replacement for EZGreetingsSubscriptionExtractJob
      migrated from XML to Spring Batch 5.2.3 Java DSL
    */

    @Bean
    public Job chatGptExtractJob() {
        logger.debug("chatGptExtractJob() called");

        return new JobBuilder("ChatGptExtractJob", jobRepository)
                .incrementer(new RunIdIncrementer())
                .start(extractStep())
                .next(decider)
                .from(decider).on("FAILED").fail()
                .from(decider).on("NO RECORDS FOUND").end()
                .from(decider).on("COMPLETED").to(encryptStep())
                .next(transferStep())
                .end()
                .build();
    }

    @Bean
    public Step extractStep() {
        logger.debug("extractStep() called");

        return new StepBuilder("extract-step1", jobRepository)
                .tasklet(extractTasklet, transactionManager)
                .taskExecutor(new SimpleAsyncTaskExecutor("extract-executor"))
                .build();
    }

    @Bean
    public Step encryptStep() {
        logger.debug("encryptStep() called");

        return new StepBuilder("encrypt-file-step", jobRepository)
                .tasklet(encryptFileTasklet, transactionManager)
                .build();
    }

    @Bean
    public Step transferStep() {
        logger.debug("transferStep() called");

                .tasklet(fileTransferTasklet, transactionManager)
                .build();
